  '한 번 더! 👍', '괜찮아요, 다시! 🌈'
];

// ===== STATE =====
let currentScreen = 0;
let starsEarned = 0;
let gameStartTime = Date.now();
let records = [];
let favorites = [];

// Step 1 state
let step1Questions = [];
let step1Index = 0;

// Step 2 state
let step2Questions = [];
let step2Index = 0;

// ===== TTS =====
function speak(text, callback) {
  if (!('speechSynthesis' in window)) { if(callback) callback(); return; }
  window.speechSynthesis.cancel();
  const u = new SpeechSynthesisUtterance(text);
  u.lang = 'ko-KR';
  u.rate = 0.8;
  u.pitch = 1.1;
  // Try to find Korean voice
  const voices = window.speechSynthesis.getVoices();
  const koVoice = voices.find(v => v.lang.startsWith('ko'));
  if (koVoice) u.voice = koVoice;
  if (callback) u.onend = callback;
  window.speechSynthesis.speak(u);
}

// Ensure voices loaded
if ('speechSynthesis' in window) {
  speechSynthesis.onvoiceschanged = () => speechSynthesis.getVoices();
}

// ===== NAVIGATION =====
function goToScreen(n) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.getElementById('screen' + n).classList.add('active');
  currentScreen = n;
  updateProgress();

  switch(n) {
    case 0: speak('안녕! 나는 뭉이야! 오늘은 여가 활동을 배워요!'); break;
    case 1: initStep1(); break;
    case 2: initStep2(); break;
    case 3: initStep3(); break;
    case 4: showSummary(); break;
  }
}

function updateProgress() {
  const pct = [0, 25, 55, 80, 100][currentScreen];
  document.getElementById('progressBar').style.width = pct + '%';
}

// ===== HELPERS =====
function shuffle(arr) {
  const a = [...arr];
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

function pick(arr, n) { return shuffle(arr).slice(0, n); }

function randomPraise() { return PRAISE[Math.floor(Math.random() * PRAISE.length)]; }
function randomRetry() { return RETRY_MSG[Math.floor(Math.random() * RETRY_MSG.length)]; }

// ===== STARS =====
function earnStar() {
  if (starsEarned >= 5) return;
  starsEarned++;
  const starEl = document.getElementById('star' + starsEarned);
  starEl.textContent = '⭐';
  starEl.classList.add('earned');
  document.getElementById('totalStars').textContent = starsEarned;
}

// ===== FEEDBACK =====
let feedbackCallback = null;

function showFeedback(emoji, text, cb) {
  document.getElementById('fbEmoji').textContent = emoji;
  document.getElementById('fbText').textContent = text;
  document.getElementById('feedbackOverlay').classList.add('show');
  feedbackCallback = cb;
  speak(text);
}

function closeFeedback() {
  document.getElementById('feedbackOverlay').classList.remove('show');
  if (feedbackCallback) feedbackCallback();
  feedbackCallback = null;
}

// ===== CELEBRATION =====
function celebrate() {
  const container = document.getElementById('confettiContainer');
  container.innerHTML = '';
  const colors = ['#FF6B6B','#4ECDC4','#45B7D1','#96CEB4','#FFEAA7','#DDA0DD','#FF9800','#66BB6A'];
  for (let i = 0; i < 60; i++) {
    const c = document.createElement('div');
    c.className = 'confetti';
    c.style.left = Math.random() * 100 + '%';
    c.style.background = colors[Math.floor(Math.random() * colors.length)];
    c.style.animationDuration = (1.5 + Math.random() * 2) + 's';
    c.style.animationDelay = (Math.random() * 0.8) + 's';
    c.style.width = (8 + Math.random() * 10) + 'px';
    c.style.height = (8 + Math.random() * 10) + 'px';
    container.appendChild(c);
  }
  setTimeout(() => container.innerHTML = '', 4000);
}

// ===== RECORDS =====
function addRecord(step, question, attempts, result, prompt) {
  records.push({ step, question, attempts, result, prompt });
  const tbody = document.getElementById('recordBody');
  const tr = document.createElement('tr');
  tr.innerHTML = `<td>${step}</td><td>${question}</td><td>${attempts}</td><td>${result}</td><td>${prompt}</td>`;
  tbody.appendChild(tr);
}

// STEP 1: 이름을 듣고 맞는 그림 고르기
// ================================================
function initStep1() {
  step1Questions = pick(ACTIVITIES, 3);
  step1Index = 0;
  showStep1Question();
}

function showStep1Question() {
  if (step1Index >= step1Questions.length) {
    earnStar();
    celebrate();
    showFeedback('🌟', '1단계 완료! 멋져요!', () => goToScreen(2));
    return;
  }

  const q = step1Questions[step1Index];
  document.getElementById('step1-target-emoji').textContent = '';
  document.getElementById('step1-target-text').textContent = '🔊 "' + q.name + '" 을 찾아요!';

  // choices: correct + 3 distractors
  let others = ACTIVITIES.filter(a => a.name !== q.name);
  let choices = shuffle([q, ...pick(others, 3)]);

  const grid = document.getElementById('step1-choices');
  grid.innerHTML = '';
  let attempts = 0;

  choices.forEach(c => {
    const card = document.createElement('div');
    card.className = 'activity-card';
    card.innerHTML = `<span class="emoji">${c.emoji}</span><span class="label">${c.name}</span>`;
    card.onclick = () => {
      attempts++;
      if (c.name === q.name) {
        card.classList.add('correct');
        const praise = randomPraise();
        earnStar();
        celebrate();
        addRecord('1단계', q.name, attempts, '성공', attempts > 1 ? '재시도' : '독립');
        speak(praise, () => {
          setTimeout(() => {
            step1Index++;
            showStep1Question();
          }, 600);
        });
        // disable all
        grid.querySelectorAll('.activity-card').forEach(c2 => c2.onclick = null);
      } else {
        card.classList.add('try-again');
        setTimeout(() => card.classList.remove('try-again'), 500);
        speak(randomRetry());
      }
    };
    grid.appendChild(card);
  });

  speakStep1Question();
}

function speakStep1Question() {
  if (step1Index < step1Questions.length) {
    const q = step1Questions[step1Index];
    speak(q.name + '을 찾아봐요!');
  }
}

// STEP 3: 좋아하는 활동 고르기
// ================================================
function initStep3() {
  favorites = [];
  document.getElementById('step3-confirm').style.display = 'none';
  const grid = document.getElementById('step3-choices');
  grid.innerHTML = '';

  ACTIVITIES.forEach(a => {
    const card = document.createElement('div');
    card.className = 'activity-card';
    card.dataset.name = a.name;
    card.innerHTML = `<span class="emoji">${a.emoji}</span><span class="label">${a.name}</span>`;
    card.onclick = () => toggleFavorite(card, a);
    grid.appendChild(card);
  });

  speak('좋아하는 여가 활동을 2개 골라봐요!');
}

function toggleFavorite(card, activity) {
  if (card.classList.contains('selected')) {
    card.classList.remove('selected');
    favorites = favorites.filter(f => f.name !== activity.name);
  } else {
    if (favorites.length >= 2) {
      speak('2개만 골라요!');
      return;
    }
    card.classList.add('selected');
    favorites.push(activity);
  }

  if (favorites.length === 2) {
    document.getElementById('step3-confirm').style.display = 'inline-flex';
  } else {
    document.getElementById('step3-confirm').style.display = 'none';
  }
}

function confirmFavorites() {
  earnStar();
  celebrate();
  addRecord('3단계', favorites.map(f=>f.name).join(', '), 1, '성공', '독립');
  document.getElementById('favRecord').textContent = favorites.map(f => f.emoji + f.name).join(', ');
  showFeedback('💖', '멋진 선택이에요!', () => goToScreen(4));
}

function showSummary() {
  const favDiv = document.getElementById('myFavorites');
  if (favorites.length > 0) {
    favDiv.innerHTML = '<p style="font-weight:700; margin-bottom:8px;">💖 내가 좋아하는 활동</p>' +
      favorites.map(f => `<span style="font-size:2.5rem; margin:0 10px;">${f.emoji}</span><br><strong>${f.name}</strong>`).join('<span style="margin:0 8px; font-size:1.5rem;">❤️</span>');
  }

  // calc time
  const elapsed = Math.round((Date.now() - gameStartTime) / 1000);
  const min = Math.floor(elapsed / 60);
  const sec = elapsed % 60;
  document.getElementById('totalTime').textContent = `${min}분 ${sec}초`;

  celebrate();
  speakSummary();
}

function speakSummary() {
  speak('정말 잘했어요! 오늘 여가 활동의 이름을 알고, 좋아하는 활동을 골랐어요! 대단해요!');
}
